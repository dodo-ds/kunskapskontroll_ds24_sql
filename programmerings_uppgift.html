emp_query = """
SELECT
    P.BusinessEntityID,
    P.FirstName,
    P.LastName,
	EMP.Gender,
    EMP.JobTitle,
    DEP.Name Department,
    EMP.HireDate,
    EMP.SalariedFlag,
	C.Country,
	C.City,
	C.Province
FROM
    Person.Person P
INNER JOIN
	(
		SELECT BAE.BusinessEntityID, PA.City, PSP.Name Province, PCR.Name Country 
		FROM Person.BusinessEntityAddress BAE,
				Person.Address PA,
				Person.StateProvince PSP,
				Person.CountryRegion PCR
		WHERE
			BAE.AddressID = PA.AddressID AND
			PA.StateProvinceID = PSP.StateProvinceID AND
			PSP.CountryRegionCode = PCR.CountryRegionCode
	) C ON C.BusinessEntityID = P.BusinessEntityID
INNER JOIN
    HumanResources.Employee EMP ON P.BusinessEntityID = EMP.BusinessEntityID
INNER JOIN
	HumanResources.EmployeeDepartmentHistory EDH ON EDH.BusinessEntityID = EMP.BusinessEntityID
INNER JOIN
	HumanResources.Department DEP ON DEP.DepartmentID = EDH.DepartmentID
WHERE 
    EDH.EndDate is NULL;
"""

# Berkna s2p
